# Note

  1. Use parentheses
  2. Use do{}while(0) or ({...}) wrap multiple statement
#define SQUARE(x) x* x
int tmp = SQUARE(3 + 3);  // tmp = 3 + 3 * 3 + 3

#define SQUARE(x) ((x) * (x))
int tmp = SQUARE(3 + 3);  // tmp = ((3 + 3) * (3 + 3))
